\begin{align*}
%% Phi
  \Phi  & ~~=~~  \phi  + \sqrt{2}\lambda\theta + F\theta^2  \\ \\
%% V  
  V  &  ~~=~~      - \overline{\theta\sigma_\mu}\theta A_\mu + 
		 i \theta^2 \overline{\theta\lambda} +
 		 i \overline{\theta^2}\theta\lambda -
          	 \frac{i}{2} \theta^2\overline{\theta}^2 D \\ \\
%% W		 
  W^\alpha & ~~=~~  -\lambda^\alpha + \theta^\alpha D  +
              \frac{1}{2}
	\left[\sigma_\mu\overline{\sigma}_\nu\theta\right]^\alpha F_{\mu\nu} -
	\theta^2 i \slashed{\mathcal{D}}^{\alpha\dot{\alpha}}
	\overline{\lambda}_{\dot{\alpha}} \\ \\
%% sigma-matrices	
  \sigma_\mu^{\alpha\dot{\alpha}} & ~~=~~ 
		\lgr 1,\; -i \tau^a \rgr~, ~~~~~~~
  \overline{\sigma}_{\mu\dot{\alpha}\alpha} ~~=~~ 
		\lgr 1,\; \phantom{+}i \tau^a \rgr \\ \\
%% Covariant derivatives -- D^\alpha
   D^\alpha &  ~~=~~  \partial^\alpha - 
            i \slashed{\partial}^{\alpha\dot{\alpha}}
	    \overline{\theta}_{\dot{\alpha}}  
\quad
   \epsilon_{\alpha\beta} ~~=~~ 
    \epsilon_{\dot{\alpha}\dot{\beta}} ~~=~~
    \lgr \begin{array}{rr}
         \!\!  0 & ~~1 \\ \\
         \!\! -1 & ~~0 
    \end{array} \rgr
\quad
	    \psi_\alpha ~~=~~ \epsilon_{\alpha\beta}\psi^\beta 
\quad
	    \overline{\psi}_{\dot{\alpha}} ~~=~~ 
			  \epsilon_{\dot{\alpha}\dot{\beta}}
			  \overline{\psi}^{\dot{\beta}}\\ 
%% Covariant derivatives -- D_{\dot \alpha}
   \overline{D}_{\dot{\alpha}} & ~~=~~ \overline{\partial}_{\dot{\alpha}} -
            i \ov{\slashed{\partial}}{}_{\dot{\alpha}\alpha}\theta^\alpha 
\quad
   \epsilon^{\alpha\beta} ~~=~~ 
    \epsilon^{\dot{\alpha}\dot{\beta}} ~~=~~
    \lgr\begin{array}{rr}
         ~0 & -1 \\ \\
         ~1 &  0
    \end{array} \rgr
\quad
	    \psi^\alpha ~~=~~ \epsilon^{\alpha\beta}\psi_\beta 
\quad
	    \overline{\psi}^{\dot{\alpha}} ~~=~~ 
			  \epsilon^{\dot{\alpha}\dot{\beta}}
			  \overline{\psi}_{\dot{\beta}}\\ \\
 \\
%%  Spinor-index contractions
  W^2 & ~~=~~  W_\alpha W^\alpha~,~~~~~~~ 
    \overline{W}^2 ~~=~~ \overline{W^{\dot{\alpha}} W_{\dot{\alpha}}} \\ \\
%%  Conjugate Field-strength
  \overline{W}_{\dot{\alpha}} & ~~=~~ 
        - \overline{\lambda}_{\dot{\alpha}} +
	  \overline{\theta}_{\dot{\alpha}} D -
	  \frac{1}{2} \left [ 
	      \overline{\sigma}_\mu\sigma_\nu \overline{\theta}
	              \right ]_{\dot{\alpha}} F_{\mu\nu} -
	  \overline{\theta}^2 (i \ov{\slashed{\mc{D}}}\lambda)_{\dot{\alpha}} \\ \\
%%  Chiral fields
  x_L & \equiv  ~~~y ~~~ 
	\equiv ~~~ x_\mu + i \overline{\theta\sigma_\mu}\theta~,~~~~~
   \Phi ~~=~~ \Phi(y)~,~~~~~~\overline{\Phi} ~~=~~ \overline{\Phi}(x_R) \\ \\
%%  Gauge derivative
  \nabla_\mu & ~~=~~  \partial_\mu - i A_\mu^a \frac{\tau^2}{2}~,~~~ 
           \text{for SU(2)} \\ \\
%%  Vector superfield in "y" variable
  V(y) & ~~=~~  - \overline{\theta\sigma_\mu}\theta A_\mu(y) + 
	    i \theta^2 \overline{\theta\lambda}(y) +
            i \overline{\theta}^2 \theta\lambda(y) -
            \frac{i}{2}\theta^2\overline{\theta}^2
	    \left [ D(y) - \partial_\mu A^\mu(y) \right ] \\ \\
%%  Definition of the field-strength
  W^\alpha & ~~=~~ - \frac{i}{4} \overline{DD} D^\alpha V 
  \qquad
  \lgr \text{has to be renormalized for $ e^{-2V} $ in the chiral representation} \rgr
	\\ \\
%%  Shift of the variable to the chiral variable (or vice versa)
  \text{Shift:} &\quad x_L ~~\to~~ x  \qquad {\rm is} \qquad e^{-i \ov{\theta\slashed{\partial}}\theta}
\end{align*}

% Definition of S and \ov{S}
\begin{align*}
%
	S &    ~~=~~    \frac{i}{2}\, \ov D{}_R D_L\, \ln \sqrt{2}\, \ov\Sigma
	&
	\ov S &    ~~=~~    \frac{i}{2}\, \ov D{}_L D_R\, \ln \sqrt{2}\, \Sigma
\end{align*}

% Components of S and \ov{S}
\begin{align*}
	S \Big| &    ~~=~~    \frac{ \sqrt{2}\ov\sigma\, ( iD \,-\, F_{03} ) ~-~ 2\,i\, \ov\lambda{}_R \lambda_L }
						{ \big( \sqrt{2}\, \ov\sigma \big)^2 }
	\quad
	&
	\ov S \Big| &    ~~=~~    \frac{ \sqrt{2}\sigma\, ( iD \,+\, F_{03} ) ~-~ 2\,i\, \ov\lambda{}_L \lambda_R }
						{ \big( \sqrt{2}\, \sigma \big)^2 }
\end{align*}

\begin{align*}
%
	D_R\, S &    ~~=~~    \p_L\, \frac{ D_L\, \sqrt{2}\,\ov\Sigma}{\sqrt{2}\,\ov\Sigma}
	&
	D_R\, S \Big| &    ~~=~~    -\, 2\, \p_L\, \frac{\ov\lambda{}_R}{\sqrt{2}\ov\sigma}
	\\[2mm]

	\\[2mm]

	\ov D{}_L S &    ~~=~~    -\, \p_R\, \frac{ \ov{D_R\, \sqrt{2}\, \Sigma} }{ \sqrt{2}\, \ov\Sigma }
